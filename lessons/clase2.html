function comparador(originalName) {
   return function (name) {
       if (name.length > originalName.length) {
           return name + ' tiene más letras que ' + originalName
       } else {
           return name + ' tiene menos letras que ' + originalName
       }
   }
}

function CountChar(list) {
   var str = " characters";
   var fns = [];

   for(var i = 0; i < list.length; i++) {
       (function (text) {
           fns.push(function(withCharText) {
               if (withCharText) {
                   console.log(text + str);
               } else {
                   console.log(text);
               }
           });
       }(list[i]));
   }

   return fns;
}

var add = function (a, b) {
    return a + b;
};

var sub = function (a, b) {
    return a - b;
};

function Calculator(init) {
   return function () {
       var operaciones = arguments;

       return function () {
           var current = init;
           for(var i = 0; i < arguments.length; i++) {
               current = operaciones[i](current, arguments[i]);
           }

           return current;
       }
   }
}
