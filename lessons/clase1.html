closures

Las variables permanecen conectadas al scope en el que fueron creadas


function holaGenerator() {
    var hola = 'hola!';

    return function () {
        return hola += 'hola!';
    }
}

var holita = holaGenerator();

console.log(holita()); //hola!hola!
console.log(holita()); //hola!hola!hola!
console.log(holita()); //hola!hola!hola!hola
